/*
R0 = 0
R1 = 128
R2 = 5
R3 = Address a
R4 = Address b
*/

ADDI R0, 128 -> R1
ADDI R0, 5 -> R2
FOR1:	STW R2 -> 0(R3)
	ADDI R3, 4 -> R3
	ADDI R1, -1 -> R1
	BEQ R1, R0 -> FOR1 (-12)
	
ADDI R0, 128 -> R1
ADDI R3, -(4*128) -> R3

FOR2:	LDW 0(R3) -> R5
	STW R5 -> 0(R4)
	ADDI R4, 4 -> R4
	BEQ R4,R0 -> FOR2 (-12)
